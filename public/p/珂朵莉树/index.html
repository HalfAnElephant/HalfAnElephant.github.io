set<Node_t>::iterator split(int x)
{
    if (x > n)
    {
        return odt.end();
    }

    auto it = --odt.upper_bound((Node_t){x, 0, 0});
    if (it->l == x)
    {
        return it;
    }

    int l = it->l;
    int r = it->r;
    int v = it->v;

    odt.erase(it);
    odt.insert(Node_t(l, x - 1, v));
    return odt.insert(Node_t(x, r, v)).first;
}
